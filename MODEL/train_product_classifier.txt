    "IMG_SIZE = (160, 160)\n",
    "BATCH_SIZE = 8\n",
    "AUTOTUNE = tf.data.AUTOTUNE\n",
    "CACHE_DATASET = False\n",
    "\n",
    "def load_and_preprocess(path, label):\n",
    "    img = tf.io.read_file(path)\n",
    "    img = tf.io.decode_image(img, channels=3, expand_animations=False)\n",
    "    img = tf.image.resize(img, IMG_SIZE)\n",
    "    img = tf.cast(img, tf.float32)\n",
    "    return img, label\n",
    "\n",
    "def make_ds(dataframe, training=False):\n",
    "    paths = dataframe['filepath'].values\n",
    "    labs = dataframe['label_idx'].values.astype(np.int32)\n",
    "    ds = tf.data.Dataset.from_tensor_slices((paths, labs))\n",
    "    ds = ds.map(load_and_preprocess, num_parallel_calls=AUTOTUNE)\n",
    "    if training:\n",
    "        ds = ds.shuffle(min(2048, len(dataframe)), seed=42)\n",
    "    if CACHE_DATASET:\n",
    "        ds = ds.cache()\n",
    "    return ds.batch(BATCH_SIZE).prefetch(1)\n",
    "\n",
    "train_ds = make_ds(train_df, training=True)\n",
    "val_ds = make_ds(val_df, training=False)\n"
