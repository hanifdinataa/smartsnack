    "def augment_image_pil(img_pil, variant):\n",
    "    \"\"\"Create augmented variants of an image using PIL.\"\"\"\n",
    "    from PIL import ImageEnhance, ImageFilter\n",
    "    import random\n",
    "\n",
    "    random.seed(variant)\n",
    "    w, h = img_pil.size\n",
    "\n",
    "    # Random horizontal flip\n",
    "    if random.random() > 0.5:\n",
    "        img_pil = img_pil.transpose(Image.FLIP_LEFT_RIGHT)\n",
    "\n",
    "    # Random rotation (-15 to +15 degrees)\n",
    "    angle = random.uniform(-15, 15)\n",
    "    img_pil = img_pil.rotate(angle, fillcolor=(128, 128, 128), expand=False)\n",
    "\n",
    "    # Random brightness\n",
    "    enhancer = ImageEnhance.Brightness(img_pil)\n",
    "    img_pil = enhancer.enhance(random.uniform(0.7, 1.3))\n",
    "\n",
    "    # Random contrast\n",
    "    enhancer = ImageEnhance.Contrast(img_pil)\n",
    "    img_pil = enhancer.enhance(random.uniform(0.7, 1.3))\n",
    "\n",
    "    # Random color/saturation\n",
    "    enhancer = ImageEnhance.Color(img_pil)\n",
    "    img_pil = enhancer.enhance(random.uniform(0.8, 1.2))\n",
    "\n",
    "    # Random crop (zoom in slightly)\n",
    "    crop_pct = random.uniform(0.85, 1.0)\n",
    "    new_w = int(w * crop_pct)\n",
    "    new_h = int(h * crop_pct)\n",
    "    left = random.randint(0, w - new_w)\n",
    "    top = random.randint(0, h - new_h)\n",
    "    img_pil = img_pil.crop((left, top, left + new_w, top + new_h))\n",
    "    img_pil = img_pil.resize((w, h), Image.LANCZOS)\n",
    "\n",
    "    # Occasional blur\n",
    "    if random.random() > 0.7:\n",
    "        img_pil = img_pil.filter(ImageFilter.GaussianBlur(radius=random.uniform(0.5, 1.5)))\n",
    "\n",
    "    return img_pil\n",

    "def extract_product_name(filename):\n",
    "    \"\"\"Extract product name from a filename.\"\"\"\n",
    "    name = Path(filename).stem\n",
    "    if name.startswith(\"IMG_\"):\n",
    "        return \"\"\n",
    "    name = re.sub(r'^\\d+_', '', name)\n",
    "    name = name.replace('_', ' ')\n",
    "    name = re.sub(r'\\s+', ' ', name).strip()\n",
    "    return name\n",

    "def organize_dataset(product_images_dir, output_dir):\n",
    "    \"\"\"\n",
    "    Organize ALL product images into class directories.\n",
    "    \n",
    "    Strategy:\n",
    "    1. Each named product file = 1 class (even if only 1 image)\n",
    "    2. Camera photos (IMG_*) are assigned to the nearest preceding named product\n",
    "    3. Classes with < MIN_AUGMENTED_IMAGES images get augmented copies\n",
    "    \"\"\"\n",
    "    print(\"=\" * 70)\n",
    "    print(\"STEP 1: ORGANIZING DATASET\")\n",
    "    print(\"=\" * 70)\n",
    "\n",
    "    # Clean output\n",
    "    if output_dir.exists():\n",
    "        shutil.rmtree(output_dir)\n",
    "    output_dir.mkdir(parents=True, exist_ok=True)\n",
    "\n",
    "    # Build class mapping\n",
    "    mapping = {}  # product_name -> {name, category, images}\n",
    "\n",
    "    for category in [\"food\", \"drink\"]:\n",
    "        cat_dir = product_images_dir / category\n",
    "        if not cat_dir.exists():\n",
    "            continue\n",
    "\n",
    "        all_files = sorted(cat_dir.iterdir())\n",
    "        current_product = None\n",
    "\n",
    "        for img_file in all_files:\n",
    "            if img_file.suffix.lower() not in ('.jpg', '.jpeg', '.png', '.bmp', '.webp'):\n",
    "                continue\n",
    "\n",
    "            product_name = extract_product_name(img_file.name)\n",
    "\n",
    "            if product_name:\n",
    "                # Named product file\n",
    "                current_product = product_name\n",
    "                if current_product not in mapping:\n",
    "                    mapping[current_product] = {\n",
    "                        \"name\": product_name,\n",
    "                        \"category\": category,\n",
    "                        \"images\": []\n",
    "                    }\n",
    "                mapping[current_product][\"images\"].append(str(img_file))\n",
    "            else:\n",
    "                # Camera photo (IMG_*) -> assign to current product\n",
    "                if current_product and current_product in mapping:\n",
    "                    mapping[current_product][\"images\"].append(str(img_file))\n",
    "\n",
    "    print(f\"\\n[OK] Found {len(mapping)} product classes\")\n",
    "\n",
    "    # Show distribution summary\n",
    "    dist = defaultdict(int)\n",
    "    for info in mapping.values():\n",
    "        n = len(info[\"images\"])\n",
    "        if n == 1:\n",
    "            dist[\"1 image\"] += 1\n",
    "        elif n <= 3:\n",
    "            dist[\"2-3 images\"] += 1\n",
    "        elif n <= 10:\n",
    "            dist[\"4-10 images\"] += 1\n",
    "        else:\n",
    "            dist[\"10+ images\"] += 1\n",
    "\n",
    "    print(\"\\nImage count distribution:\")\n",
    "    for k, v in sorted(dist.items()):\n",
    "        print(f\"  {k}: {v} classes\")\n",
    "\n",
    "    # Create directories and copy/augment images\n",
    "    class_count = 0\n",
    "    total_images = 0\n",
    "    total_augmented = 0\n",
    "\n",
    "    for key in sorted(mapping.keys()):\n",
    "        info = mapping[key]\n",
    "        safe_name = re.sub(r'[<>:\"/\\\\|?*]', '_', info[\"name\"])\n",
    "        class_dir = output_dir / safe_name\n",
    "        class_dir.mkdir(parents=True, exist_ok=True)\n",
    "\n",
    "        # Copy and sanitize original images (convert WebP/invalid formats to JPEG)\n",
    "        original_count = 0\n",
    "        for idx, img_path in enumerate(info[\"images\"]):\n",
    "            src = Path(img_path)\n",
    "            dst = class_dir / f\"orig_{idx:04d}.jpg\"  # Always save as proper JPEG\n",
    "            try:\n",
    "                # Re-save through PIL to ensure valid JPEG format\n",
    "                # (some .jpg files are actually WebP which TF can't decode)\n",
    "                img = Image.open(str(src)).convert('RGB')\n",
    "                img.save(str(dst), 'JPEG', quality=95)\n",
    "                original_count += 1\n",
    "            except Exception as e:\n",
    "                print(f\"  [WARN] Could not process {src.name}: {e}\")\n",
    "\n",
    "        if original_count == 0:\n",
    "            shutil.rmtree(str(class_dir), ignore_errors=True)\n",
    "            continue\n",
    "\n",
    "        # Augment if needed\n",
    "        aug_count = 0\n",
    "        if original_count < MIN_AUGMENTED_IMAGES:\n",
    "            needed = MIN_AUGMENTED_IMAGES - original_count\n",
    "            orig_files = list(class_dir.glob(\"orig_*\"))\n",
    "\n",
    "            for aug_idx in range(needed):\n",
    "                # Pick a random source image (cyclic)\n",
    "                src_file = orig_files[aug_idx % len(orig_files)]\n",
    "                try:\n",
    "                    img = Image.open(str(src_file)).convert('RGB')\n",
    "                    augmented = augment_image_pil(img, aug_idx + 1000)\n",
    "                    dst = class_dir / f\"aug_{aug_idx:04d}.jpg\"\n",
    "                    augmented.save(str(dst), quality=90)\n",
    "                    aug_count += 1\n",
    "                except Exception as e:\n",
    "                    print(f\"  [WARN] Augmentation failed for {safe_name}: {e}\")\n",
    "\n",
    "        class_count += 1\n",
    "        total_images += original_count + aug_count\n",
    "        total_augmented += aug_count\n",
    "\n",
    "    print(f\"\\n[OK] Created {class_count} class directories\")\n",
    "    print(f\"[OK] Total images: {total_images} (original + {total_augmented} augmented)\")\n",
    "    print(f\"[OK] Dataset saved to: {output_dir}\")\n",
    "\n",
    "    # Print top classes\n",
    "    print(\"\\nTop 20 classes by image count:\")\n",
    "    sorted_items = sorted(mapping.items(), key=lambda x: len(x[1][\"images\"]), reverse=True)\n",
    "    for i, (key, info) in enumerate(sorted_items[:20]):\n",
    "        n_orig = len(info[\"images\"])\n",
    "        n_aug = max(0, MIN_AUGMENTED_IMAGES - n_orig)\n",
    "        print(f\"  [{info['category'][:1].upper()}] {info['name']}: {n_orig} orig + {n_aug} aug = {n_orig + n_aug}\")\n",
    "\n",
    "    if len(sorted_items) > 20:\n",
    "        print(f\"  ... and {len(sorted_items) - 20} more classes\")\n",
    "\n",
    "    return mapping\n",
    "\n",
    "\n",
    "# ============================================================================\n",
    "# STEP 2: BUILD AND TRAIN MODEL\n",
    "# ============================================================================\n",
    "\n",
    "def create_data_augmentation():\n",
    "    \"\"\"Runtime data augmentation (on top of offline augmentation).\"\"\"\n",
    "    return keras.Sequential([\n",
    "        layers.RandomFlip(\"horizontal\"),\n",
    "        layers.RandomRotation(0.1),\n",
    "        layers.RandomZoom((-0.1, 0.1)),\n",
    "        layers.RandomBrightness(0.15),\n",
    "        layers.RandomContrast(0.15),\n",
    "    ], name=\"data_augmentation\")\n",
    "\n",
    "\n",
    "def build_model(num_classes):\n",
    "    \"\"\"Build MobileNetV2-based model.\"\"\"\n",
    "    base_model = MobileNetV2(\n",
    "        input_shape=(IMG_SIZE, IMG_SIZE, 3),\n",
    "        include_top=False,\n",
    "        weights='imagenet'\n",
    "    )\n",
    "    base_model.trainable = False\n",
    "\n",
    "    inputs = keras.Input(shape=(IMG_SIZE, IMG_SIZE, 3))\n",
    "    x = create_data_augmentation()(inputs, training=True)\n",
    "    x = preprocess_input(x)\n",
    "    x = base_model(x, training=False)\n",
    "    x = layers.GlobalAveragePooling2D()(x)\n",
    "    x = layers.Dense(512, activation='relu',\n",
    "                     kernel_regularizer=keras.regularizers.l2(0.001))(x)\n",
    "    x = layers.Dropout(0.4)(x)\n",
    "    x = layers.Dense(256, activation='relu',\n",
    "                     kernel_regularizer=keras.regularizers.l2(0.001))(x)\n",
    "    x = layers.Dropout(0.3)(x)\n",
    "    outputs = layers.Dense(num_classes, activation='softmax')(x)\n",
    "\n",
    "    model = keras.Model(inputs, outputs)\n",
    "    return model, base_model\n",
    "\n",
    "\n",
    "def load_datasets(dataset_dir):\n",
    "    \"\"\"Load training and validation datasets.\"\"\"\n",
    "    print(\"\\nLoading datasets...\")\n",
    "\n",
    "    train_ds = keras.utils.image_dataset_from_directory(\n",
    "        str(dataset_dir),\n",
    "        validation_split=VALIDATION_SPLIT,\n",
    "        subset=\"training\",\n",
    "        seed=42,\n",
    "        image_size=(IMG_SIZE, IMG_SIZE),\n",
    "        batch_size=BATCH_SIZE,\n",
    "        label_mode='categorical',\n",
    "        shuffle=True,\n",
    "    )\n",
    "\n",
    "    val_ds = keras.utils.image_dataset_from_directory(\n",
    "        str(dataset_dir),\n",
    "        validation_split=VALIDATION_SPLIT,\n",
    "        subset=\"validation\",\n",
    "        seed=42,\n",
    "        image_size=(IMG_SIZE, IMG_SIZE),\n",
    "        batch_size=BATCH_SIZE,\n",
    "        label_mode='categorical',\n",
    "        shuffle=False,\n",
    "    )\n",
    "\n",
    "    class_names = train_ds.class_names\n",
    "    print(f\"[OK] {len(class_names)} classes loaded\")\n",
    "    print(f\"[OK] Training batches: {len(train_ds)}\")\n",
    "    print(f\"[OK] Validation batches: {len(val_ds)}\")\n",
    "\n",
    "    AUTOTUNE = tf.data.AUTOTUNE\n",
    "    train_ds = train_ds.prefetch(buffer_size=1)\n",
    "    val_ds = val_ds.prefetch(buffer_size=1)\n",
    "\n",
    "    return train_ds, val_ds, class_names\n",
    "\n",
    "\n",
    "def train_model(dataset_dir, output_dir):\n",
    "    \"\"\"Two-phase training pipeline.\"\"\"\n",
    "    print(\"\\n\" + \"=\" * 70)\n",
    "    print(\"STEP 2: TRAINING MODEL\")\n",
    "    print(\"=\" * 70)\n",
    "\n",
    "    output_dir.mkdir(parents=True, exist_ok=True)\n",
    "    train_ds, val_ds, class_names = load_datasets(dataset_dir)\n",
    "    num_classes = len(class_names)\n",
    "\n",
    "    print(f\"\\nBuilding MobileNetV2 model for {num_classes} classes...\")\n",
    "    model, base_model = build_model(num_classes)\n",
    "\n",
    "    # === PHASE 1: Head only ===\n",
    "    print(\"\\n\" + \"-\" * 50)\n",
    "    print(f\"PHASE 1: Training head only ({EPOCHS_PHASE1} epochs)\")\n",
    "    print(\"-\" * 50)\n",
    "\n",
    "    model.compile(\n",
    "        optimizer=keras.optimizers.Adam(learning_rate=LEARNING_RATE_PHASE1),\n",
    "        loss='categorical_crossentropy',\n",
    "        metrics=['accuracy']\n",
    "    )\n",
    "\n",
    "    model.summary()\n",
    "\n",
    "    h1 = model.fit(\n",
    "        train_ds, validation_data=val_ds,\n",
    "        epochs=EPOCHS_PHASE1,\n",
    "        callbacks=[\n",
    "            callbacks.EarlyStopping(monitor='val_accuracy', patience=5,\n",
    "                                    restore_best_weights=True, verbose=1),\n",
    "            callbacks.ReduceLROnPlateau(monitor='val_loss', factor=0.5,\n",
    "                                       patience=3, min_lr=1e-6, verbose=1),\n",
    "        ],\n",
    "        verbose=1\n",
    "    )\n",
    "\n",
    "    p1_acc = max(h1.history.get('val_accuracy', [0]))\n",
    "    print(f\"\\n[OK] Phase 1 best val_accuracy: {p1_acc:.4f}\")\n",
    "\n",
    "    # === PHASE 2: Fine-tune (optional) ===\n",
    "    if EPOCHS_PHASE2 > 0:\n",
    "        print(\"\\n\" + \"-\" * 50)\n",
    "        print(f\"PHASE 2: Fine-tuning ({EPOCHS_PHASE2} epochs)\")\n",
    "        print(\"-\" * 50)\n",
    "\n",
    "        base_model.trainable = True\n",
    "        for layer in base_model.layers[:FINE_TUNE_FROM_LAYER]:\n",
    "            layer.trainable = False\n",
    "\n",
    "        trainable = sum(1 for l in base_model.layers if l.trainable)\n",
    "        print(f\"[OK] Unfroze {trainable} layers for fine-tuning\")\n",
    "\n",
    "        model.compile(optimizer=keras.optimizers.Adam(learning_rate=LEARNING_RATE_PHASE2),\n",
    "                      loss='categorical_crossentropy', metrics=['accuracy'])\n",
    "\n",
    "        h2 = model.fit(train_ds, validation_data=val_ds, epochs=EPOCHS_PHASE2,\n",
    "                       callbacks=[\n",
    "                           callbacks.EarlyStopping(monitor='val_accuracy', patience=7, restore_best_weights=True, verbose=1),\n",
    "                           callbacks.ReduceLROnPlateau(monitor='val_loss', factor=0.5, patience=3, min_lr=1e-7, verbose=1),\n",
